def count_base_dna(dna, base):
    """
    Counts the number of occurence of `base` (str) in `dna` (str)
    """
    i = 0 # counter
    for char in dna:
        if char == base:
            i += 1
    return i

#!/usr/bin/env python3
# -*- coding: utf-8 -*-

"""
Voici un exemple de code qui prend la valeur du signal Is mesur pour une gamme talon 
dont les concentrations initiales Cs sont connues. Ensuite, on calcule l'absorbance dfinie comme :
$$A = - \log\left(\dfrac{I}{I_0}\right) $$ o I0 est l'absorbance du blanc. 

Ayant deux points, on extrait ensuite la droite affine passant par ces deux points 
pour avoir la rponse du dtecteur.

Code propos par Martin Vrot sous licence CC-BY-NC-SA
"""

import numpy as np

def calc_absorbance(I,I0):
    """
    Calcul de l'absorbance
    - I est l'intensit lumineuse en sortie de l'chantillon (float)
    - I0 est l'intensit lumineuse pour le blanc (float)
    retourne l'absorbance (float)
    """
    return -np.log(I/I0)

def calc_droite(Abs,Cs):
    """
    Calcul de la pente et l'ordonne  l'origine  partir de deux points
    - Abs : liste des absorbances de longueur 2
    - Cs : liste des concentrations de longueur 2
    retourne la pente et l'ordonne  l'origine (tuple)
    """
    a = (Abs[1]-Abs[0])/(Cs[1]-Cs[0])
    b = Abs[1]-a*Cs[1]
    return a,b #slope, intercept
