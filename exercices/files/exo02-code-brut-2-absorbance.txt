#!/usr/bin/env python3
# -*- coding: utf-8 -*-

import numpy as np


def calc_absorbance(I, I0):
    return -np.log(I / I0)


def calc_droite(Abs, Cs):
    a = (Abs[1] - Abs[0]) / (Cs[1] - Cs[0])
    b = Abs[1] - a * Cs[1]
    return a, b


if __name__ == "__main__":
    Is = [611, 281]
    Abs = []
    Cs = [2, 4]
    I0 = 986
    for I in Is:
        Abs.append(calc_absorbance(I, I0))
    droite = calc_droite(Abs, Cs)
    print("pente : {}, ordonnée à l'origine : {}".format(*droite))
